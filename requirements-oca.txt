# Módulos OCA instalados como wheels de PyPI dentro de la imagen de Odoo.
# Se instalan con --no-deps (la dependencia "odoo" del wheel no es pip-instalable
# en la imagen oficial); las dependencias python reales ya están en la imagen.
#
# Fase 0.3 — ejecución asíncrona del agente:
odoo-addon-queue-job==18.0.*
# Fase 0.4 — traza independiente de los cambios hechos vía agente:
odoo-addon-auditlog==18.0.*
# Ola 2 — extensiones OCA de CRM (tools condicionales: solo se activan si el
# módulo está instalado en la BD):
odoo-addon-crm-lead-code==18.0.*
odoo-addon-crm-lead-product==18.0.*
odoo-addon-crm-phonecall==18.0.*
odoo-addon-partner-identification==18.0.*
# Ola 3 — ventas (tipos de pedido, excepciones, aprobaciones por niveles):
odoo-addon-sale-order-type==18.0.*
odoo-addon-sale-exception==18.0.*
# Dependencia de sale_exception:
odoo-addon-base-exception==18.0.*
odoo-addon-base-tier-validation==18.0.*
odoo-addon-sale-tier-validation==18.0.*
# Ola 4 — almacén (stock sin reservar, ajustes de inventario agrupados):
odoo-addon-stock-available-unreserved==18.0.*
odoo-addon-stock-inventory==18.0.*
# Ola 5 — compras (solicitudes internas y tipos de pedido).
# NOTA: purchase_discount NO tiene wheel 18.0 publicado -> pospuesto (gate).
odoo-addon-purchase-request==18.0.*
odoo-addon-purchase-order-type==18.0.*
# Ola 6 — finanzas (remesas/órdenes de pago):
odoo-addon-account-payment-order==18.0.*
# Dependencias de account_payment_order:
odoo-addon-account-payment-partner==18.0.*

# Ola 7 — helpdesk (alternativa gratuita al Helpdesk Enterprise):
odoo-addon-helpdesk-mgmt==18.0.*
